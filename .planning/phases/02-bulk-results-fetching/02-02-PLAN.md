Create `components/BulkResultsButton.tsx` as a new file with the following complete implementation:

```typescript
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";

export type FetchStatus = "idle" | "pending" | "success" | "not_ready" | "error";

interface GameStatus {
  game_id: string;
  date: string;
  game_type: string;
  track: string;
  status: FetchStatus;
  message?: string;
}

interface Props {
  pendingGames: Array<{
    game_id: string;
    date: string;
    game_type: string;
    track: string;
  }>;
}

function delay(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

export function BulkResultsButton({ pendingGames }: Props) {
  const router = useRouter();
  const [fetching, setFetching] = useState(false);
  const [statuses, setStatuses] = useState<GameStatus[]>([]);

  async function handleFetchAll() {
    if (pendingGames.length === 0 || fetching) return;

    setFetching(true);
    // Initialise all rounds as "pending" so the user sees the list immediately
    setStatuses(
      pendingGames.map((g) => ({
        game_id: g.game_id,
        date: g.date,
        game_type: g.game_type,
        track: g.track,
        status: "pending",
      }))
    );

    for (let i = 0; i < pendingGames.length; i++) {
      const game = pendingGames[i];

      // 500 ms delay between requests (skip before the very first one)
      if (i > 0) await delay(500);

      try {
        const res = await fetch(
          `/api/games/${encodeURIComponent(game.game_id)}/results`,
          { method: "POST" }
        );
        const data = await res.json().catch(() => ({}));

        let nextStatus: FetchStatus;
        let message: string | undefined;

        if (res.status === 422) {
          nextStatus = "not_ready";
          message = "Inte redo";
        } else if (res.ok) {
          nextStatus = "success";
          message = `${data.updated ?? "?"} hästar i ${data.races ?? "?"} avd`;
        } else {
          nextStatus = "error";
          message = data.error ?? `HTTP ${res.status}`;
        }

        setStatuses((prev) =>
          prev.map((s) =>
            s.game_id === game.game_id
              ? { ...s, status: nextStatus, message }
              : s
          )
        );
      } catch (err) {
        setStatuses((prev) =>
          prev.map((s) =>
            s.game_id === game.game_id
              ? {
                  ...s,
                  status: "error",
                  message: err instanceof Error ? err.message : "Nätverksfel",
                }
              : s
          )
        );
      }
    }

    setFetching(false);
    router.refresh();
  }

  const pendingCount = pendingGames.length;

  return (
    <div className="flex flex-col gap-3">
      <button
        onClick={handleFetchAll}
        disabled={pendingCount === 0 || fetching}
        className="self-start px-4 py-2 rounded-lg border border-gray-300 dark:border-gray-600 text-gray-700 dark:text-gray-300 text-sm font-semibold hover:border-indigo-500 dark:hover:border-indigo-400 hover:text-indigo-600 dark:hover:text-indigo-400 disabled:opacity-40 transition"
      >
        {fetching
          ? "Hämtar..."
          : pendingCount === 0
          ? "Alla resultat hämtade"
          : `Hämta alla resultat (${pendingCount})`}
      </button>

      {statuses.length > 0 && (
        <div className="flex flex-col gap-1">
          {statuses.map((s) => (
            <div
              key={s.game_id}
              className="flex items-center justify-between px-4 py-2 rounded-lg bg-gray-100 dark:bg-gray-800 text-sm"
            >
              <span className="text-gray-700 dark:text-gray-300">
                {s.date} · {s.game_type} · {s.track}
              </span>
              <span
                className={`flex items-center gap-1.5 text-xs font-semibold px-2 py-0.5 rounded-full ${
                  s.status === "success"
                    ? "bg-green-100 dark:bg-green-900/40 text-green-700 dark:text-green-400"
                    : s.status === "not_ready"
                    ? "bg-yellow-100 dark:bg-yellow-900/40 text-yellow-700 dark:text-yellow-400"
                    : s.status === "error"
                    ? "bg-red-100 dark:bg-red-900/40 text-red-700 dark:text-red-400"
                    : "bg-gray-200 dark:bg-gray-700 text-gray-500 dark:text-gray-400"
                }`}
              >
                {s.status === "pending" && "Väntar..."}
                {s.status === "success" && `Klar — ${s.message}`}
                {s.status === "not_ready" && "Inte redo"}
                {s.status === "error" && `Fel: ${s.message}`}
              </span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser client → Next.js API route | BulkResultsButton makes POST requests to `/api/games/{gameId}/results` — gameId originates from the server-rendered allGames list (not free-form user input) |
| Next.js API route → ATG external API | Existing results route calls `fetchGameResults()` from `lib/atg.ts`; this boundary is unchanged and outside this plan's scope |
| Next.js API route → Supabase | Results route uses `createServiceClient()` (service role key) to update finish_position — unchanged, not modified in this plan |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Tampering | `BulkResultsButton.tsx` — gameId in fetch URL | mitigate | gameIds come exclusively from `allGames` prop derived from a server-side Supabase query. They are UUID strings from the database. The existing results route validates `gameId` and checks the game exists before processing (`if (!game) return 404`). No additional client-side validation required — trust boundary is at the API layer. |
| T-02-02-02 | Denial of Service | Sequential fetch loop — 500ms delay | accept | The 500ms delay (EVAL-03) mitigates ATG rate limiting. The `games` table query in Plan 02-01 is limited to 100 rows. Maximum burst is therefore 100 requests over 50 seconds — acceptable for a single authenticated admin. |
| T-02-02-03 | Elevation of Privilege | `/api/games/{gameId}/results` — no auth check in route handler | accept | The results route uses `createServiceClient()` (service role) but does not validate the caller's session. This is a pre-existing condition not introduced by this plan. Mitigation path: add `createClient()` session check in the route handler — deferred to a future security hardening phase. Risk is low: the route only updates finish_position, not user/group data, and the app requires auth at the layout level. |
| T-02-02-04 | Information Disclosure | Error messages from ATG forwarded to client | accept | `data.error` from the API is rendered in the status badge (`Fel: ${data.error}`). ATG error messages are not expected to contain PII. If they ever do, truncate to 100 chars: `message.slice(0, 100)`. Acceptable for current scope. |
</threat_model>
