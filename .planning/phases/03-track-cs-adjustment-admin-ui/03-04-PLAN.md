    export default async function AuthenticatedLayout({
      children,
    }: {
      children: React.ReactNode;
    }) {
      const supabase = await createClient();
      const { data: { user } } = await supabase.auth.getUser();

      const adminIds = (process.env.ADMIN_USER_IDS ?? "")

        .map((s) => s.trim())
        .filter(Boolean);
      const isAdmin = user != null && adminIds.includes(user.id);

          <TopNav />
          {children}
          <BottomNav isAdmin={isAdmin} />
          <InstallPrompt />
        </div>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Layout → BottomNav (client) | isAdmin is computed server-side and serialized as a boolean prop — client cannot override it |
| Admin link visibility | Visibility is cosmetic — the real authorization gate is the server-side redirect in admin/page.tsx. Hiding the link from non-admins is UX only. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-04-01 | Elevation of Privilege | isAdmin prop passed to BottomNav | accept | isAdmin is a display hint, not an authorization check. Real authorization is in admin/page.tsx redirect. Even if a user fabricates the prop via browser devtools, they cannot reach admin content — they would only see a nav link that leads to a redirect. |
| T-3-04-02 | Information Disclosure | Admin tab presence reveals admin status | accept | Knowing you are an admin is not sensitive. ADMIN_USER_IDS is an env var that controls this — it is not exposed to clients. |
</threat_model>
